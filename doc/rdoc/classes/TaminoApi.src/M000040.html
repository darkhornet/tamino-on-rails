  def setHttpRequestMethod(sMethod)
    
    if ( (sMethod != "GET") && (sMethod != "POST") )
      raise TaminoAPIException ,  "The Http Method is not recognized: \"#{sMethod}\""
    end
    
    if (sMethod == "GET") 
      @_sHttpRequestMethod = "GET"
    elsif sMethod == "POST"
      @_sHttpRequestMethod = "POST"
    else
      return false
    end
    return true      
  end
