  def undefine(sSchema)
    
    unless sSchema.is_a? String
      raise TaminoAPIException , "The Schema is not recognized: \"#{sSchema}\""
    end
    @_bReturnValue = true
    _sendRequest(@_sCollection, TAMINO_OP_UNDEFINE, sSchema)
    getHeaderResponse
    return @_bReturnValue
  end
